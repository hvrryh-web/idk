### HP Bar Generation
```typescript
function generatePokemonHPBar(current: number, max: number, width: number = 20): string {
  const percentage = current / max;
  const filled = Math.floor(percentage * width);
  const empty = width - filled;
  
  let char = 'â–“'; // green
  if (percentage < 0.5) char = 'â–’'; // yellow
  if (percentage < 0.25) char = 'â–‘'; // red
  
  return char.repeat(filled) + ' '.repeat(empty);
}
```
